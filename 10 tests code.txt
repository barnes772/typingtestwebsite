const passages = [
  "The quick brown fox jumps over the lazy dog.",
  "Lorem ipsum dolor sit amet, consectetur adipiscing elit.",
  "Aenean imperdiet hendrerit dui, eget faucibus felis.",
  "Suspendisse potenti. Nulla volutpat elementum massa id pellentesque.",
  "In hac habitasse platea dictumst. Sed finibus eleifend urna, eget faucibus odio tristique ut.",
  "Vivamus feugiat sagittis quam, vel varius eros aliquam in.",
  "Pellentesque habitant morbi tristique senectus et netus et malesuada fames ac turpis egestas.",
  "Proin commodo turpis sit amet lacus consequat, at lacinia sapien sollicitudin.",
  "Phasellus blandit fringilla faucibus. Sed venenatis metus sed mauris sollicitudin, ac blandit mauris bibendum.",
  "Nulla commodo, sapien eget placerat commodo, quam nisl finibus eros, sit amet auctor elit sapien sed elit."
];

function generateTypingTest() {
  const typingTest = [];

  while (typingTest.length < 10) {
    const randomPassageIndex = Math.floor(Math.random() * passages.length);
    const randomPassage = passages[randomPassageIndex];

    if (!typingTest.includes(randomPassage)) {
      typingTest.push(randomPassage);
    }
  }

  return typingTest;
}

// Example usage:
const typingTest = generateTypingTest();
console.log(typingTest);
