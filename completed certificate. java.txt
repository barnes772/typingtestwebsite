// Get the customer's name and typing speed
const customerName = "John Doe";
const typingSpeed = 60;

// Generate the certificate text
const certificateText = `Certificate of Typing Proficiency
This is to certify that ${customerName} has completed a typing test and achieved a typing speed of ${typingSpeed} words per minute.

Congratulations on your achievement!

Signed,
The Typing Test Team`;

// Create a new document and add the certificate text
const doc = new jsPDF();
doc.setFontSize(16);
doc.text(certificateText, 10, 10);

// Save the document as a PDF file and download it
doc.save(`${customerName}_typing_certificate.pdf`);
